function getZodiac(date) {
  let d = new Date(date);
  let day = d.getDate();
  let month = d.getMonth() + 1;

  if((month==3 && day>=21)||(month==4 && day<=19)) return "Aries";
  if((month==4 && day>=20)||(month==5 && day<=20)) return "Taurus";
  if((month==5 && day>=21)||(month==6 && day<=20)) return "Gemini";
  if((month==6 && day>=21)||(month==7 && day<=22)) return "Cancer";
  if((month==7 && day>=23)||(month==8 && day<=22)) return "Leo";
  if((month==8 && day>=23)||(month==9 && day<=22)) return "Virgo";
  if((month==9 && day>=23)||(month==10 && day<=22)) return "Libra";
  if((month==10 && day>=23)||(month==11 && day<=21)) return "Scorpio";
  if((month==11 && day>=22)||(month==12 && day<=21)) return "Sagittarius";
  if((month==12 && day>=22)||(month==1 && day<=19)) return "Capricorn";
  if((month==1 && day>=20)||(month==2 && day<=18)) return "Aquarius";
  return "Pisces";
}

function typeText(text) {
  let i = 0;
  let speed = 40;
  let resultDiv = document.getElementById("result");
  resultDiv.innerHTML = "";

  function typing() {
    if (i < text.length) {
      resultDiv.innerHTML += text.charAt(i);
      i++;
      setTimeout(typing, speed);
    }
  }
  typing();
}

function calculateLove() {
  let name1 = document.getElementById("yourName").value;
  let name2 = document.getElementById("partnerName").value;
  let dob1 = document.getElementById("yourDOB").value;
  let dob2 = document.getElementById("partnerDOB").value;

  if(!name1 || !name2 || !dob1 || !dob2) {
    document.getElementById("result").innerHTML = "⚠️ Fill all details";
    return;
  }

  let zodiac1 = getZodiac(dob1);
  let zodiac2 = getZodiac(dob2);
  let lovePercent = Math.floor(Math.random()*100)+1;

  let message = `✨ ${zodiac1} ❤️ ${zodiac2}\n💘 Love: ${lovePercent}%`;
  typeText(message);

  document.getElementById("proposalBox").style.display = "block";
}

function createHearts() {
  for(let i=0;i<20;i++){
    let heart = document.createElement("div");
    heart.className = "heart";
    heart.innerHTML = "❤️";
    heart.style.left = Math.random()*100 + "%";
    heart.style.top = "80%";
    document.body.appendChild(heart);

    setTimeout(()=>{ heart.remove(); },2000);
  }
}

function answerYes() {
  let yourName = document.getElementById("yourName").value;
  let partnerName = document.getElementById("partnerName").value;

  if(!yourName || !partnerName){
    alert("Enter names first ❤️");
    return;
  }

  let finalMessage = `💍 ${yourName} proposed to ${partnerName}! ❤️\n🎉 ${partnerName} accepted! 💖`;

  // Save history
  let history = JSON.parse(localStorage.getItem("loveHistory") || "[]");
  history.push(finalMessage);
  localStorage.setItem("loveHistory", JSON.stringify(history));

  // Show typing result
  typeText(finalMessage);

  // Heart animation
  createHearts();

  // WhatsApp share after delay
  setTimeout(() => {
    let url = "https://wa.me/?text=" + encodeURIComponent(finalMessage);
    window.open(url, '_blank');
  }, 1500);
}

function moveNo() {
  let btn = document.getElementById("noBtn");
  btn.style.position = "absolute";
  btn.style.top = Math.random()*300 + "px";
  btn.style.left = Math.random()*300 + "px";
}
